#include <stdio.h>

int factorial(int);

int main() {
    int num;

    scanf("%d", &num);
    num = factorial(num);
    printf("%d", num);

    return 0;
}

int factorial(int num) {
    if(num == 0) // 재귀함수 빠져나올 수 있는 조건
        return 1;
    
    return num * factorial(num - 1);
}

num = 5
